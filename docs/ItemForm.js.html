import React from "react";

/**
 * Komponenta formuláře pro položky.
 *
 * Tato komponenta umožňuje uživateli zadat údaje o položce a odeslat je pomocí tlačítka.
 *
 * @param {Object} props
 * @param {string} props.inputName - Název položky.
 * @param {function} props.setInputName - Funkce pro změnu názvu položky.
 * @param {string} props.inputType - Typ položky.
 * @param {function} props.setInputType - Funkce pro změnu typu položky.
 * @param {number} props.inputQuantity - Počet položek.
 * @param {function} props.setInputQuantity - Funkce pro změnu počtu položek.
 * @param {string} props.inputPurchased - Datum nákupu položky.
 * @param {function} props.setInputPurchased - Funkce pro změnu data nákupu položky.
 * @param {boolean} props.inputCanExpire - Možnost expirace položky.
 * @param {function} props.setInputCanExpire - Funkce pro změnu možnosti expirace položky.
 * @param {function} props.handleSubmit - Funkce pro odeslání formuláře.
 * @param {boolean} props.isEditing - Stav editace položky.
 * @returns {JSX.Element}
 */
export default function ItemForm({
  inputName,
  setInputName,
  inputType,
  setInputType,
  inputQuantity,
  setInputQuantity,
  inputPurchased,
  setInputPurchased,
  inputCanExpire,
  setInputCanExpire,
  handleSubmit,
  isEditing,
}) {
  return (
    <form className="form">
      <input
        type="text"
        placeholder="Název..."
        value={inputName}
        onChange={(e) => setInputName(e.target.value)}
        required
      />
      <select
        placeholder="Typ..."
        value={inputType}
        onChange={(e) => setInputType(e.target.value)}
        required
      >
        <option value="">Typ...</option>
        <option value="Ovoce">Ovoce</option>
        <option value="Těstoviny">Těstoviny</option>
        <option value="Zelenina">Zelenina</option>
        <option value="Pití">Pití</option>
      </select>
      <input
        type="number"
        placeholder="Pocet..."
        value={inputQuantity}
        onChange={(e) => setInputQuantity(e.target.value)}
        required
      />
      <input
        type="date"
        placeholder="Datum..."
        value={inputPurchased}
        onChange={(e) => setInputPurchased(e.target.value)}
        required
      />
      <label htmlFor="canExpire">Expiruje?</label>
      <input
        className="form-check-input"
        id="canExpire"
        type="checkbox"
        checked={inputCanExpire}
        onChange={(e) => setInputCanExpire(e.target.checked)}
      />
      <button type="submit" onClick={handleSubmit} className={isEditing ? "btn btn-warning" : "btn btn-primary"}>
        {isEditing ? "Uložit" : "Přidat"}
      </button>
    </form>
  );
}
